ANSWERS

QUESTION 1
Display the project names where the number of employees working on it is 4 and order by project name


select proj_name,count(*) from project p, employee e where p.dep_id = e.dep_id group by proj_id having count(*)=4 order by proj_name;


QUESTION 2
Display the names of the clerks who work on ProjectZ and have been hired after the year 2000.


select emp_name from project p,employee e where proj_name = 'ProjectZ'and post='Clerk' and year(hire_date)>2000 and p.dep_id = e.dep_id;


QUESTION 3
Display all the unique job groups in the descending order.


select distinct post from employee order by post desc;


QUESTION 4
List all the employees who joined in the month of april or may in descending order of seniority.


select emp_id, emp_name, hire_date from employee where hire_date like '%-04-%' or hire_date like '%-05-%' order by hire_date desc;


QUESTION 5
Display the names,id and salary of the clerks and departments who have no comission and work in ABC block.


select emp_name, emp_id, salary from employee e, department d where dep_location = 'ABC Block' and e.dep_id = d.dep_id and comission IS NULL;
